## 背景

### ZGC 对象迁移

```
ZGC 迁移过程:

1. 标记阶段: 识别存活对象
2. 迁移阶段: 将存活对象复制到新位置
3. 更新引用: 更新所有指向旧位置的引用

传统迁移:
┌─────────────────────────────────────────────────────────────┐
│ 源页面 (任何 NUMA 节点)                                      │
│ [对象A] [对象B] [对象C] [对象D] [对象E] [对象F]               │
└─────────────────────────────────────────────────────────────┘
                        │
                        ▼ 迁移 (不考虑 NUMA)
┌─────────────────────────────────────────────────────────────┐
│ 目标页面 (可能是任何 NUMA 节点)                              │
│ [对象A] [对象B] [对象C] [对象D] [对象E] [对象F]               │
└─────────────────────────────────────────────────────────────┘

问题: 对象可能被迁移到远离访问线程的 NUMA 节点
```

### NUMA-Aware 迁移

```
NUMA-Aware 迁移:

┌─────────────────────────────────────────────────────────────┐
│ 源页面 (混合访问模式)                                        │
│ [A:Node0] [B:Node1] [C:Node0] [D:Node1] [E:Node0] [F:Node1] │
└─────────────────────────────────────────────────────────────┘
                        │
                        ▼ 分析访问模式
        ┌───────────────┴───────────────┐
        ▼                               ▼
┌───────────────────┐         ┌───────────────────┐
│ 目标页面 (Node 0)  │         │ 目标页面 (Node 1)  │
│ [对象A] [对象C]    │         │ [对象B] [对象D]    │
│ [对象E]           │         │ [对象F]           │
│ (Node 0 线程访问)  │         │ (Node 1 线程访问)  │
└───────────────────┘         └───────────────────┘

优势: 对象靠近访问它的线程，减少跨节点访问
```

```cpp
// 文件: src/hotspot/share/gc/z/zNUMA.cpp

#include "gc/z/zNUMA.hpp"
#include "runtime/os.hpp"
#include "runtime/thread.hpp"
#include "utilities/debug.hpp"

bool ZNUMA::_enabled = false;
int ZNUMA::_num_nodes = 1;
int* ZNUMA::_node_to_cpu_map = nullptr;
int* ZNUMA::_cpu_to_node_map = nullptr;

void ZNUMA::initialize() {
    if (!os::numa_has_static_binding()) {
        _enabled = false;
        _num_nodes = 1;
        return;
    }
    
    _enabled = true;
    _num_nodes = os::numa_get_num_nodes();
    
    // 构建 CPU <-> Node 映射
    int max_cpu = os::active_processor_count();
    _cpu_to_node_map = NEW_C_HEAP_ARRAY(int, max_cpu, mtGC);
    _node_to_cpu_map = NEW_C_HEAP_ARRAY(int, _num_nodes, mtGC);
    
    for (int cpu = 0; cpu < max_cpu; cpu++) {
        _cpu_to_node_map[cpu] = -1;
    }
    
    for (int node = 0; node < _num_nodes; node++) {
        int cpus[256];
        int count = os::numa_get_node_cpus(node, cpus, 256);
        
        if (count > 0) {
            _node_to_cpu_map[node] = cpus[0];
            for (int i = 0; i < count; i++) {
                _cpu_to_node_map[cpus[i]] = node;
            }
        }
    }
    
    log_info(gc, numa)("ZGC NUMA enabled: %d nodes", _num_nodes);
}

int ZNUMA::cpu_to_node(int cpu) {
    if (!_enabled || cpu < 0) {
        return 0;
    }
    return _cpu_to_node_map[cpu];
}

int ZNUMA::node_to_first_cpu(int node) {
    if (!_enabled || node < 0 || node >= _num_nodes) {
        return 0;
    }
    return _node_to_cpu_map[node];
}

int ZNUMA::current_thread_node() {
    if (!_enabled) {
        return 0;
    }
    
    Thread* thread = Thread::current();
    if (thread == nullptr) {
        return 0;
    }
    
    // 方法 1: 使用 JVM 内部 API
    int node = os::get_thread_numa_node();
    if (node >= 0) {
        return node;
    }
    
    // 方法 2: 通过 CPU 推断
    int cpu = os::processor_id();
    return cpu_to_node(cpu);
}

int ZNUMA::analyze_object_numa_preference(oop obj) {
    if (!_enabled) {
        return 0;
    }
    
    // 简化实现: 返回当前线程的 NUMA 节点
    // 完整实现需要追踪对象访问历史
    return current_thread_node();
}

void* ZNUMA::alloc_on_node(size_t size, int node) {
    if (!_enabled || node < 0 || node >= _num_nodes) {
        return os::malloc(size, mtGC);
    }
    
    // 设置 NUMA 内存策略
    os::set_numa_preferred_node(node);
    
    // 分配内存
    void* mem = os::malloc(size, mtGC);
    
    // 恢复默认策略
    // (实际实现可能不需要恢复，因为只影响当前线程)
    
    return mem;
}
```

### ZPage NUMA 标记

```cpp
// 文件: src/hotspot/share/gc/z/zPage.hpp

class ZPage {
private:
    // ... 原有字段 ...
    
    // 新增: NUMA 节点标记
    int _numa_node;
    
    // 新增: 访问统计 (用于分析)
    uint64_t _access_counts[ZNUMA::max_nodes()];
    
public:
    // ... 原有方法 ...
    
    // ===== NUMA 相关 =====
    
    /**
     * 获取页面的 NUMA 节点
     */
    int numa_node() const { return _numa_node; }
    
    /**
     * 设置页面的 NUMA 节点
     */
    void set_numa_node(int node) { _numa_node = node; }
    
    /**
     * 记录访问 (用于 NUMA 分析)
     */
    void record_access(int node) {
        if (node >= 0 && node < ZNUMA::max_nodes()) {
            Atomic::inc(&_access_counts[node]);
        }
    }
    
    /**
     * 获取主要访问节点的 NUMA 节点
     */
    int get_primary_access_node() const {
        int max_node = 0;
        uint64_t max_count = _access_counts[0];
        
        for (int i = 1; i < ZNUMA::num_nodes(); i++) {
            if (_access_counts[i] > max_count) {
                max_count = _access_counts[i];
                max_node = i;
            }
        }
        
        return max_node;
    }
};
```

### NUMA-Aware 迁移逻辑

```cpp
// 文件: src/hotspot/share/gc/z/zRelocate.cpp

#include "gc/z/zRelocate.hpp"
#include "gc/z/zNUMA.hpp"
#include "gc/z/zPage.hpp"
#include "gc/z/zForwarding.hpp"

class ZRelocate {
    
    // ... 原有方法 ...
    
    /**
     * NUMA-Aware 对象迁移
     */
    zaddress relocate_object_numa_aware(zaddress from_addr) {
        // 1. 获取对象的 NUMA 偏好
        int preferred_node = analyze_object_numa_preference(from_addr);
        
        // 2. 选择目标页面
        ZPage* target_page = select_target_page_numa_aware(preferred_node);
        
        // 3. 在目标页面分配空间
        zaddress to_addr = target_page->allocate_object(from_addr);
        
        if (to_addr != zaddress::null) {
            // 4. 复制对象
            copy_object(from_addr, to_addr);
            
            // 5. 更新统计
            record_relocation(from_addr, target_page->numa_node());
        }
        
        return to_addr;
    }
    
    /**
     * 分析对象的 NUMA 偏好
     */
    int analyze_object_numa_preference(zaddress addr) {
        if (!ZNUMA::is_enabled()) {
            return 0;
        }
        
        // 方法 1: 使用页面访问统计
        ZPage* source_page = page(addr);
        if (source_page != nullptr) {
            int primary_node = source_page->get_primary_access_node();
            if (primary_node >= 0) {
                return primary_node;
            }
        }
        
        // 方法 2: 使用当前线程的 NUMA 节点
        return ZNUMA::current_thread_node();
    }
    
    /**
     * 选择 NUMA-Aware 目标页面
     */
    ZPage* select_target_page_numa_aware(int preferred_node) {
        if (!ZNUMA::is_enabled()) {
            return select_target_page_default();
        }
        
        // 优先选择相同 NUMA 节点的页面
        ZPage* page = find_page_on_node(preferred_node);
        
        if (page != nullptr && page->has_space()) {
            return page;
        }
        
        // 如果没有合适的页面，创建新页面
        return allocate_page_on_node(preferred_node);
    }
    
    /**
     * 在指定 NUMA 节点上分配页面
     */
    ZPage* allocate_page_on_node(int node) {
        // 设置内存分配策略
        os::set_numa_preferred_node(node);
        
        // 分配页面
        ZPage* page = allocate_page(ZPageType::small);
        
        if (page != nullptr) {
            page->set_numa_node(node);
        }
        
        return page;
    }
};
```

class ZForwarding {

    struct RelocationStats {
        int objects_migrated;
        int bytes_migrated;
        int cross_node_migrations;  // 跨节点迁移数
    } _stats;
    
public:
    // ... 原有方法 ...
    
    /**
     * 获取目标 NUMA 节点
     */
    int target_numa_node() const { return _target_numa_node; }
    
    /**
     * 设置目标 NUMA 节点
     */
    void set_target_numa_node(int node) { 
        _target_numa_node = node; 
    }
    
    /**
     * 记录迁移
     */
    void record_migration(int from_node, int to_node, size_t size) {
        _stats.objects_migrated++;
        _stats.bytes_migrated += size;
        
        if (from_node != to_node) {
            _stats.cross_node_migrations++;
        }
    }
    
    /**
     * 获取迁移统计
     */
    const RelocationStats& stats() const { return _stats; }

## 工作原理详解

### 对象访问追踪

```
对象访问追踪机制:

┌─────────────────────────────────────────────────────────────┐
│                     对象访问                                 │
└─────────────────────────────────────────────────────────────┘
                        │
                        ▼
┌─────────────────────────────────────────────────────────────┐
│  读屏障 (Load Barrier)                                       │
│  if (object_needs_numa_tracking(obj)) {                     │
│      int node = ZNUMA::current_thread_node();               │
│      page(obj)->record_access(node);                        │
│  }                                                          │
└─────────────────────────────────────────────────────────────┘
                        │
                        ▼
┌─────────────────────────────────────────────────────────────┐
│  页面访问统计                                                │
│  Page A: {Node0: 1000, Node1: 200}  → 主要访问: Node0       │
│  Page B: {Node0: 100, Node1: 800}   → 主要访问: Node1       │
│  Page C: {Node0: 500, Node1: 500}   → 均匀访问              │
└─────────────────────────────────────────────────────────────┘
                        │
                        ▼ GC 迁移
┌─────────────────────────────────────────────────────────────┐
│  NUMA-Aware 迁移决策                                        │
│  Page A → 迁移到 Node0                                      │
│  Page B → 迁移到 Node1                                      │
│  Page C → 保持或随机选择                                    │
└─────────────────────────────────────────────────────────────┘
```

### 迁移算法

```cpp
/**
 * NUMA-Aware 迁移算法
 */
void ZRelocate::relocate_page_numa_aware(ZPage* page) {
    // 1. 分析页面的访问模式
    int primary_node = page->get_primary_access_node();
    
    // 2. 如果访问模式均匀，跳过 NUMA 优化
    if (is_uniform_access(page)) {
        relocate_page_default(page);
        return;
    }
    
    // 3. 获取或创建目标 NUMA 节点的页面
    ZPage* target = get_or_create_page_on_node(primary_node);
    
    // 4. 遍历源页面的所有对象
    for (zaddress obj : page->objects()) {
        // 4.1 分析单个对象的访问模式
        int obj_node = analyze_object_access_pattern(obj);
        
        // 4.2 选择目标
        ZPage* obj_target = (obj_node == primary_node) 
                          ? target 
                          : get_or_create_page_on_node(obj_node);
        
        // 4.3 迁移对象
        zaddress new_addr = obj_target->allocate_object(obj);
        copy_object(obj, new_addr);
        
        // 4.4 更新转发表
        forwarding_table()->insert(obj, new_addr);
    }
    
    // 5. 记录统计
    log_numa_migration(page, target);
}
```
