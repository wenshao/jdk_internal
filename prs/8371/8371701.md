# NUMA 线程亲和性

> JDK-8371701 | Joel Sikström | JDK 26
> **Note**: Joel Sikström is a Software Engineer at Oracle on the HotSpot GC team, focusing on ZGC improvements.

---

## 概述

本 PR 新增 API 允许设置线程的 **NUMA (Non-Uniform Memory Access) 亲和性**，优化多插槽服务器的内存访问性能。通过将线程绑定到特定 NUMA 节点，减少跨节点内存访问延迟。

## 背景

### NUMA 架构

```
多插槽服务器 NUMA 架构:

┌─────────────────────────────────────────────────────────────────┐
│                        系统总线 (QPI/UPI)                        │
├──────────────────────────┬──────────────────────────────────────┤
│      NUMA Node 0         │          NUMA Node 1                 │
│  ┌────────────────────┐  │  ┌────────────────────┐              │
│  │    CPU 0-15        │  │  │    CPU 16-31       │              │
│  │    (Socket 0)      │  │  │    (Socket 1)      │              │
│  ├────────────────────┤  │  ├────────────────────┤              │
│  │   Memory 0         │  │  │   Memory 1         │              │
│  │   (本地内存)        │  │  │   (本地内存)        │              │
│  └────────────────────┘  │  └────────────────────┘              │
└──────────────────────────┴──────────────────────────────────────┘

内存访问延迟:
- 本地访问 (Node 0 → Memory 0): ~100ns
- 远程访问 (Node 0 → Memory 1): ~200-300ns (2-3x 慢!)
```

### 问题分析

```
无 NUMA 亲和性的问题:

┌─────────────────────────────────────────────────────┐
│  线程在 Node 0 运行                                  │
│  分配的内存在 Node 1                                 │
│                                                     │
│  ┌──────────┐              ┌──────────┐            │
│  │  线程    │ ─跨节点访问─→ │  内存    │            │
│  │ (Node 0) │   (慢!)      │ (Node 1) │            │
│  └──────────┘              └──────────┘            │
└─────────────────────────────────────────────────────┘

有 NUMA 亲和性的优化:

┌─────────────────────────────────────────────────────┐
│  线程绑定到 Node 0                                   │
│  内存优先从 Node 0 分配                              │
│                                                     │
│  ┌──────────┐              ┌──────────┐            │
│  │  线程    │ ─本地访问──→ │  内存    │            │
│  │ (Node 0) │   (快!)      │ (Node 0) │            │
│  └──────────┘              └──────────┘            │
└─────────────────────────────────────────────────────┘
```

### Linux 实现

```cpp
// 文件: src/hotspot/os/linux/os_linux.cpp

#include <sched.h>
#include <numa.h>
#include <numaif.h>

bool os::numa_has_static_binding() {
    return numa_available() >= 0;
}

int os::numa_get_num_nodes() {
    if (numa_available() < 0) {
        return 1;
    }
    return numa_max_node() + 1;
}

int os::numa_get_node_cpus(int node, int* cpus, size_t size) {
    if (numa_available() < 0 || node < 0) {
        return -1;
    }
    
    struct bitmask* cpumask = numa_allocate_cpumask();
    if (cpumask == nullptr) {
        return -1;
    }
    
    int count = 0;
    if (numa_node_to_cpus(node, cpumask) == 0) {
        for (int cpu = 0; cpu < cpumask->size && count < (int)size; cpu++) {
            if (numa_bitmask_isbitset(cpumask, cpu)) {
                cpus[count++] = cpu;
            }
        }
    }
    
    numa_free_cpumask(cpumask);
    return count;
}

bool os::set_thread_numa_affinity(int node) {
    if (numa_available() < 0 || node < 0 || node > numa_max_node()) {
        return false;
    }
    
    // 方法 1: 使用 CPU 亲和性
    struct bitmask* cpumask = numa_allocate_cpumask();
    if (cpumask == nullptr) {
        return false;
    }
    
    if (numa_node_to_cpus(node, cpumask) != 0) {
        numa_free_cpumask(cpumask);
        return false;
    }
    
    // 转换为 cpu_set_t
    cpu_set_t set;
    CPU_ZERO(&set);
    
    for (int cpu = 0; cpu < (int)cpumask->size; cpu++) {
        if (numa_bitmask_isbitset(cpumask, cpu)) {
            CPU_SET(cpu, &set);
        }
    }
    
    numa_free_cpumask(cpumask);
    
    // 设置线程亲和性
    // 注意: sched_setaffinity(0, ...) 表示当前线程
    int result = sched_setaffinity(0, sizeof(set), &set);
    
    if (result == 0) {
        // 同时设置内存分配策略
        set_numa_preferred_node(node);
    }
    
    return result == 0;
}

int os::get_thread_numa_node() {
    if (numa_available() < 0) {
        return -1;
    }
    
    // 方法 1: 通过 CPU 亲和性推断
    cpu_set_t set;
    CPU_ZERO(&set);
    
    if (sched_getaffinity(0, sizeof(set), &set) != 0) {
        return -1;
    }
    
    // 找到第一个 CPU 所在的 NUMA 节点
    for (int cpu = 0; cpu < CPU_SETSIZE; cpu++) {
        if (CPU_ISSET(cpu, &set)) {
            // 查询 CPU 属于哪个 NUMA 节点
            for (int node = 0; node <= numa_max_node(); node++) {
                struct bitmask* cpumask = numa_allocate_cpumask();
                if (cpumask != nullptr) {
                    numa_node_to_cpus(node, cpumask);
                    if (numa_bitmask_isbitset(cpumask, cpu)) {
                        numa_free_cpumask(cpumask);
                        return node;
                    }
                    numa_free_cpumask(cpumask);
                }
            }
        }
    }
    
    // 方法 2: 通过内存分配推断
    void* page = nullptr;
    int node = -1;
    
    // 分配一个页面
    page = mmap(nullptr, 4096, PROT_READ | PROT_WRITE,
                MAP_PRIVATE | MAP_ANONYMOUS, -1, 0);
    if (page != MAP_FAILED) {
        // 查询页面所在的 NUMA 节点
        if (get_mempolicy(&node, nullptr, 0, page, MPOL_F_NODE | MPOL_F_ADDR) == 0) {
            // 成功获取
        }
        munmap(page, 4096);
    }
    
    return node;
}

void os::set_numa_preferred_node(int node) {
    if (numa_available() < 0 || node < 0) {
        return;
    }
    
    // 设置内存分配策略为 MPOL_PREFERRED
    // 新分配的内存优先从指定节点分配
    struct bitmask* nodemask = numa_allocate_nodemask();
    if (nodemask != nullptr) {
        numa_bitmask_setbit(nodemask, node);
        set_mempolicy(MPOL_PREFERRED, nodemask->maskp, nodemask->size);
        numa_free_nodemask(nodemask);
    }
}
```

### Windows 实现

```cpp
// 文件: src/hotspot/os/windows/os_windows.cpp

#include <windows.h>
#include <processthreadsapi.h>

bool os::numa_has_static_binding() {
    // Windows 从 Vista 开始支持 NUMA
    return true;
}

int os::numa_get_num_nodes() {
    ULONG highest_node;
    if (GetNumaHighestNodeNumber(&highest_node)) {
        return highest_node + 1;
    }
    return 1;
}

int os::numa_get_node_cpus(int node, int* cpus, size_t size) {
    ULONGLONG processor_mask;
    if (!GetNumaNodeProcessorMask(node, &processor_mask)) {
        return -1;
    }
    
    int count = 0;
    for (int cpu = 0; cpu < 64 && count < (int)size; cpu++) {
        if (processor_mask & (1ULL << cpu)) {
            cpus[count++] = cpu;
        }
    }
    
    return count;
}

bool os::set_thread_numa_affinity(int node) {
    // Windows 使用处理器组
    GROUP_AFFINITY affinity = {0};
    
    ULONGLONG processor_mask;
    if (!GetNumaNodeProcessorMask(node, &processor_mask)) {
        return false;
    }
    
    affinity.Mask = (KAFFINITY)processor_mask;
    // 设置处理器组 (对于 >64 CPU 的系统)
    affinity.Group = 0; // 简化处理
    
    HANDLE thread = GetCurrentThread();
    return SetThreadGroupAffinity(thread, &affinity, nullptr);
}

int os::get_thread_numa_node() {
    // Windows 没有直接获取线程 NUMA 节点的 API
    // 通过处理器组推断
    PROCESSOR_NUMBER proc_number;
    GetCurrentProcessorNumberEx(&proc_number);
    
    // 查询处理器属于哪个 NUMA 节点
    USHORT node;
    if (GetNumaProcessorNode(&proc_number, &node)) {
        return node;
    }
    
    return -1;
}
```

## 使用场景

### 1. 高性能计算 (HPC)

```java
// 场景: 大规模矩阵运算
// 每个工作线程处理不同 NUMA 节点的数据

public class MatrixMultiplication {
    
    private final int numaNodes;
    
    public MatrixMultiplication() {
        this.numaNodes = ThreadAffinity.getNumaNodeCount();
    }
    
    public void multiply(double[][] A, double[][] B, double[][] C) {
        ExecutorService executor = Executors.newFixedThreadPool(numaNodes);
        int rowsPerNode = A.length / numaNodes;
        
        List<Future<?>> futures = new ArrayList<>();
        
        for (int node = 0; node < numaNodes; node++) {
            final int startRow = node * rowsPerNode;
            final int endRow = (node == numaNodes - 1) 
                             ? A.length 
                             : (node + 1) * rowsPerNode;
            final int numaNode = node;
            
            futures.add(executor.submit(() -> {
                // 绑定到指定 NUMA 节点
                ThreadAffinity.setNumaAffinity(numaNode);
                
                // 计算本地数据
                for (int i = startRow; i < endRow; i++) {
                    for (int j = 0; j < B[0].length; j++) {
                        for (int k = 0; k < A[0].length; k++) {
                            C[i][j] += A[i][k] * B[k][j];
                        }
                    }
                }
            }));
        }
        
        // 等待所有任务完成
        for (Future<?> f : futures) {
            f.get();
        }
        
        executor.shutdown();
    }
}
```

### 2. 数据库系统

```java
// 场景: 分区数据库
// 每个分区绑定到特定 NUMA 节点

public class PartitionedDatabase {
    
    private final Map<Integer, Partition> partitions;
    private final int numaNodes;
    
    public PartitionedDatabase(int partitionCount) {
        this.numaNodes = ThreadAffinity.getNumaNodeCount();
        this.partitions = new HashMap<>();
        
        // 创建分区，均匀分布到 NUMA 节点
        for (int i = 0; i < partitionCount; i++) {
            int node = i % numaNodes;
            partitions.put(i, new Partition(i, node));
        }
    }
    
    public void handleQuery(int partitionId, Query query) {
        Partition partition = partitions.get(partitionId);
        int numaNode = partition.getNumaNode();
        
        // 在正确的 NUMA 节点上执行查询
        executor.submit(() -> {
            ThreadAffinity.setNumaAffinity(numaNode);
            partition.execute(query);
        });
    }
}
```

### 3. 缓存系统

```java
// 场景: 分片缓存
// 每个分片的内存优先从本地 NUMA 节点分配

public class ShardedCache<K, V> {
    
    private final CacheShard<K, V>[] shards;
    private final int numaNodes;
    
    @SuppressWarnings("unchecked")
    public ShardedCache(int shardCount) {
        this.numaNodes = ThreadAffinity.getNumaNodeCount();
        this.shards = new CacheShard[shardCount];
        
        for (int i = 0; i < shardCount; i++) {
            int node = i % numaNodes;
            
            // 在目标 NUMA 节点上创建分片
            ThreadAffinity.setNumaAffinity(node);
            shards[i] = new CacheShard<>(i, node);
        }
    }
    
    public V get(K key) {
        int shardId = key.hashCode() % shards.length;
        int node = shards[shardId].getNumaNode();
        
        // 在数据所在的 NUMA 节点上执行
        ThreadAffinity.setNumaAffinity(node);
        return shards[shardId].get(key);
    }
    
    public void put(K key, V value) {
        int shardId = key.hashCode() % shards.length;
        int node = shards[shardId].getNumaNode();
        
        ThreadAffinity.setNumaAffinity(node);
        shards[shardId].put(key, value);
    }
}
```

## 注意事项

### 限制

1. **需要 root 或 CAP_SYS_NICE 权限**才能设置线程亲和性
2. **不适用于虚拟化环境** (NUMA 拓扑可能被隐藏)
3. **容器环境需要正确配置** (--cpuset-cpus 等)

### 最佳实践

```java
// ✅ 正确: 在线程启动时设置
Thread worker = new Thread(() -> {
    ThreadAffinity.setNumaAffinity(node);
    // 工作...
});
worker.start();

// ❌ 错误: 频繁切换亲和性
for (int i = 0; i < 1000; i++) {
    ThreadAffinity.setNumaAffinity(i % numaNodes);
    // 工作...  // 性能反而下降!
}

// ✅ 正确: 根据数据位置设置
int dataNode = getDataNumaNode(data);
ThreadAffinity.setNumaAffinity(dataNode);
processData(data);
```
