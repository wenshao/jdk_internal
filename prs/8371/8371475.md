# HttpClient CUBIC 拥塞控制

> JDK-8371475 | Daniel Jeliński | JDK 26

---

## 概述

本 PR 为 HTTP/3 (QUIC) 实现了 **CUBIC 拥塞控制算法**，替代原有的 Reno 算法。CUBIC 是现代 TCP 的标准拥塞控制算法，特别适合高延迟高带宽网络环境。

## 背景

### 拥塞控制算法对比

```
Reno (传统算法):
                    拥塞窗口
                       │
        慢启动         │    拥塞避免 (线性)
          │           │         │
          ▼           │         ▼
          ┌───────────┼─────────┐
          │          /│        /│
          │         / │       / │
          │        /  │      /  │
          │       /   │     /   │
          │      /    │    /    │
          │     /     │   /     │
          └────┴──────┴──┴──────┴───► 时间
                   丢包

问题: 高延迟网络下，线性恢复太慢


CUBIC (现代算法):
                    拥塞窗口
                       │
                      /│\
                     / │ \
                    /  │  \     快速恢复
                   /   │   \   /
                  /    │    \ /
                 /     │     X
                /      │    / \
               /       │   /   \
              ─────────┼──┴─────┴───► 时间
                       │
                 三次函数曲线

优势: 快速恢复到最优窗口
```

### 算法原理

**CUBIC 窗口计算公式:**

```
W(t) = C × (t - k)³ + W_max

其中:
- W(t): 时刻 t 的拥塞窗口
- C: 常数 (0.4)
- t: 距上次拥塞的时间
- k: 窗口增长到 W_max 的时间
- W_max: 上次拥塞时的窗口大小

k = ³√(W_max × β / C)

β = 0.7 (乘法减少因子)
```

```java
// 文件: QuicCongestionController.java

package jdk.internal.net.http;

/**
 * QUIC 拥塞控制器接口
 */
public interface QuicCongestionController {
    
    /**
     * 获取当前拥塞窗口大小
     */
    long getCongestionWindow();
    
    /**
     * 获取慢启动阈值
     */
    long getSlowStartThreshold();
    
    /**
     * 收到 ACK 时调用
     * @param ackedBytes 确认的字节数
     * @param rtt 当前 RTT
     */
    void onAckReceived(long ackedBytes, long rtt);
    
    /**
     * 检测到丢包时调用
     */
    void onPacketLoss();
    
    /**
     * 发生拥塞事件时调用
     * @param cause 拥塞原因
     */
    void onCongestionEvent(CongestionCause cause);
    
    /**
     * 是否可以发送数据
     * @param bytesInFlight 已发送未确认的字节数
     */
    boolean canSend(long bytesInFlight);
    
    /**
     * 获取发送节奏延迟
     */
    long getPacingDelay();
    
    enum CongestionCause {
        PACKET_LOSS,      // 丢包
        ECN,              // 显式拥塞通知
        TIMEOUT           // 超时
    }
}
```

### 基类实现

```java
// 文件: QuicBaseCongestionController.java (318 行)

package jdk.internal.net.http;

import java.util.concurrent.atomic.AtomicLong;

/**
 * QUIC 拥塞控制器基类
 * 实现通用的慢启动和 ACK 处理逻辑
 */
abstract class QuicBaseCongestionController implements QuicCongestionController {
    
    // ===== 常量 =====
    protected static final long INITIAL_WINDOW = 10 * 1200; // 10 个最大包
    protected static final long MINIMUM_WINDOW = 2 * 1200;  // 2 个最大包
    protected static final long MAX_DATAGRAM_SIZE = 1200;   // QUIC 最大包
    
    // ===== 状态变量 =====
    protected final AtomicLong cwnd = new AtomicLong(INITIAL_WINDOW);
    protected final AtomicLong ssthresh = new AtomicLong(Long.MAX_VALUE);
    protected final AtomicLong bytesInFlight = new AtomicLong(0);
    
    // ===== RTT 测量 =====
    protected volatile long smoothedRtt = 100_000_000; // 100ms (纳秒)
    protected volatile long rttVariance = 50_000_000;  // 50ms
    protected volatile long minRtt = Long.MAX_VALUE;
    
    // ===== 发送节奏 =====
    protected final QuicPacer pacer;
    
    protected QuicBaseCongestionController(QuicPacer pacer) {
        this.pacer = pacer;
    }
    
    // ===== 接口实现 =====
    
    @Override
    public long getCongestionWindow() {
        return cwnd.get();
    }
    
    @Override
    public long getSlowStartThreshold() {
        return ssthresh.get();
    }
    
    @Override
    public void onAckReceived(long ackedBytes, long rtt) {
        // 更新 RTT 统计
        updateRttStats(rtt);
        
        // 更新 bytes in flight
        bytesInFlight.addAndGet(-ackedBytes);
        
        if (cwnd.get() < ssthresh.get()) {
            // ===== 慢启动阶段 =====
            // 指数增长: 每收到一个 ACK，cwnd 增加 ackedBytes
            cwnd.addAndGet(ackedBytes);
            
            if (cwnd.get() >= ssthresh.get()) {
                // 进入拥塞避免阶段
                onEnterCongestionAvoidance();
            }
        } else {
            // ===== 拥塞避免阶段 =====
            // 由子类实现具体算法
            onCongestionAvoidance(ackedBytes, rtt);
        }
        
        // 更新发送节奏
        pacer.update(getCongestionWindow(), smoothedRtt);
    }
    
    @Override
    public void onPacketLoss() {
        // 快速重传/快速恢复
        long currentCwnd = cwnd.get();
        long newSsthresh = computeSsthresh(currentCwnd);
        
        ssthresh.set(newSsthresh);
        cwnd.set(newSsthresh);
        
        // 通知子类
        onLossDetected();
    }
    
    @Override
    public boolean canSend(long inFlight) {
        return inFlight < cwnd.get();
    }
    
    // ===== 抽象方法 =====
    
    /**
     * 拥塞避免阶段的窗口增长
     * 由子类实现
     */
    protected abstract void onCongestionAvoidance(long ackedBytes, long rtt);
    
    /**
     * 检测到丢包时的处理
     * 由子类实现
     */
    protected abstract void onLossDetected();
    
    /**
     * 进入拥塞避免阶段
     */
    protected void onEnterCongestionAvoidance() {
        // 默认实现: 无操作
    }
    
    /**
     * 计算新的 ssthresh
     */
    protected abstract long computeSsthresh(long currentCwnd);
    
    // ===== 辅助方法 =====
    
    private void updateRttStats(long rtt) {
        // 更新最小 RTT
        if (rtt < minRtt) {
            minRtt = rtt;
        }
        
        // 更新平滑 RTT (EWMA)
        // SRTT = (7/8) * SRTT + (1/8) * RTT
        long delta = rtt - smoothedRtt;
        smoothedRtt += delta / 8;
        
        // 更新 RTT 方差
        if (delta < 0) delta = -delta;
        rttVariance += (delta - rttVariance) / 4;
    }
    
    @Override
    public long getPacingDelay() {
        return pacer.getPacingDelay();
    }
}
```

### CUBIC 实现

```java
// 文件: QuicCubicCongestionController.java (170 行)

package jdk.internal.net.http;

/**
 * CUBIC 拥塞控制算法实现
 * 
 * RFC 8312: CUBIC for Fast Long-Distance Networks
 * 
 * 窗口增长函数: W(t) = C × (t - k)³ + W_max
 */
final class QuicCubicCongestionController extends QuicBaseCongestionController {
    
    // ===== CUBIC 参数 =====
    
    /** CUBIC 常数 C */
    private static final double C = 0.4;
    
    /** 乘法减少因子 β */
    private static final double BETA = 0.7;
    
    /** TCP 友好性因子 α */
    private static final double ALPHA = 3 * (1 - BETA) / (1 + BETA);
    
    // ===== CUBIC 状态 =====
    
    /** 上次拥塞事件时的窗口大小 */
    private volatile long wMax;
    
    /** 窗口增长到 wMax 的时间 (纳秒) */
    private volatile double k;
    
    /** 上次拥塞事件的时间 */
    private volatile long epochStart;
    
    /** 上次窗口更新的时间 */
    private volatile long lastUpdateTime;
    
    /** 累积的 ACK 字节数 */
    private final AtomicLong ackAccumulator = new AtomicLong(0);
    
    // ===== 构造函数 =====
    
    QuicCubicCongestionController(QuicPacer pacer) {
        super(pacer);
        this.wMax = 0;
        this.k = 0;
        this.epochStart = 0;
        this.lastUpdateTime = 0;
    }
    
    // ===== 核心算法 =====
    
    /**
     * 计算当前 CUBIC 窗口
     */
    private long computeCubicWindow(long currentTime) {
        if (epochStart == 0) {
            // 首次运行或重置后
            return cwnd.get();
        }
        
        // 计算经过的时间 (秒)
        double t = (currentTime - epochStart) / 1_000_000_000.0;
        
        // CUBIC 窗口计算: W(t) = C × (t - k)³ + W_max
        double w = C * Math.pow(t - k, 3) + wMax;
        
        // 确保窗口不小于最小值
        return Math.max((long) w, MINIMUM_WINDOW);
    }
    
    /**
     * 计算 TCP 友好窗口
     * 确保 CUBIC 不会比 TCP Reno 更差
     */
    private long computeTcpFriendlyWindow(long currentTime) {
        if (epochStart == 0) {
            return cwnd.get();
        }
        
        double t = (currentTime - epochStart) / 1_000_000_000.0;
        
        // TCP Reno 窗口: W_tcp = α × t + W_max × β
        double wTcp = ALPHA * t + wMax * BETA;
        
        return Math.max((long) wTcp, MINIMUM_WINDOW);
    }
    
    // ===== 实现抽象方法 =====
    
    @Override
    protected void onCongestionAvoidance(long ackedBytes, long rtt) {
        long currentTime = System.nanoTime();
        
        // 累积 ACK
        ackAccumulator.addAndGet(ackedBytes);
        
        // 时间窗口更新 (每 RTT 更新一次)
        if (currentTime - lastUpdateTime < smoothedRtt) {
            return;
        }
        lastUpdateTime = currentTime;
        
        // 计算 CUBIC 窗口
        long wCubic = computeCubicWindow(currentTime);
        
        // 计算 TCP 友好窗口
        long wTcp = computeTcpFriendlyWindow(currentTime);
        
        // 取较大值
        long newCwnd = Math.max(wCubic, wTcp);
        
        // 考虑累积的 ACK
        long accumulated = ackAccumulator.getAndSet(0);
        long increment = newCwnd - cwnd.get();
        
        if (increment > 0) {
            // 按比例增加窗口
            long increase = (accumulated * increment) / cwnd.get();
            cwnd.addAndGet(increase);
        }
    }
    
    @Override
    protected void onLossDetected() {
        long currentTime = System.nanoTime();
        long currentCwnd = cwnd.get();
        
        // 保存当前窗口作为 W_max
        wMax = currentCwnd;
        
        // 计算 k = ³√(W_max × β / C)
        k = Math.cbrt(wMax * (1 - BETA) / C);
        
        // 重置 epoch
        epochStart = currentTime;
        lastUpdateTime = currentTime;
        
        // 重置累积器
        ackAccumulator.set(0);
    }
    
    @Override
    protected long computeSsthresh(long currentCwnd) {
        // CUBIC 乘法减少: ssthresh = cwnd × β
        return (long) (currentCwnd * BETA);
    }
    
    @Override
    protected void onEnterCongestionAvoidance() {
        // 进入拥塞避免时，初始化 CUBIC 状态
        if (epochStart == 0) {
            wMax = cwnd.get();
            k = 0;
            epochStart = System.nanoTime();
            lastUpdateTime = epochStart;
        }
    }
    
    // ===== 调试和监控 =====
    
    /**
     * 获取当前 W_max
     */
    public long getWMax() {
        return wMax;
    }
    
    /**
     * 获取 k 值 (秒)
     */
    public double getK() {
        return k / 1_000_000_000.0;
    }
    
    @Override
    public String toString() {
        return String.format(
            "CUBIC[cwnd=%d, ssthresh=%d, wMax=%d, k=%.3fs]",
            cwnd.get(), ssthresh.get(), wMax, getK()
        );
    }
}
```

### 发送节奏控制

```java
// 文件: QuicPacer.java

package jdk.internal.net.http;

/**
 * QUIC 发送节奏控制器
 * 平滑数据发送，避免突发
 */
final class QuicPacer {
    
    // 节奏因子 (1.25 表示比理想速率快 25%)
    private static final double PACING_GAIN = 1.25;
    
    private volatile long pacingRate = 0;     // 字节/秒
    private volatile long nextSendTime = 0;   // 下次发送时间
    
    /**
     * 更新节奏参数
     */
    void update(long cwnd, long smoothedRtt) {
        // 节奏速率 = cwnd / RTT × 增益
        if (smoothedRtt > 0) {
            pacingRate = (long) (cwnd * 1_000_000_000L * PACING_GAIN / smoothedRtt);
        }
    }
    
    /**
     * 获取发送延迟
     * @return 发送前应等待的纳秒数
     */
    long getPacingDelay() {
        if (pacingRate == 0) {
            return 0;
        }
        
        long now = System.nanoTime();
        if (now < nextSendTime) {
            return nextSendTime - now;
        }
        return 0;
    }
    
    /**
     * 记录发送
     * @param bytes 发送的字节数
     */
    void onPacketSent(long bytes) {
        if (pacingRate > 0) {
            // 计算这次发送应该延迟的时间
            long delay = bytes * 1_000_000_000L / pacingRate;
            nextSendTime = Math.max(System.nanoTime(), nextSendTime) + delay;
        }
    }
}
```

```java
// 文件: test/jdk/java/net/http/QuicCubicCongestionControllerTest.java

import org.testng.annotations.Test;
import static org.testng.Assert.*;

public class QuicCubicCongestionControllerTest {
    
    @Test
    public void testInitialWindow() {
        QuicPacer pacer = new QuicPacer();
        QuicCubicCongestionController cc = 
            new QuicCubicCongestionController(pacer);
        
        // 初始窗口应为 10 个最大包
        assertEquals(cc.getCongestionWindow(), 10 * 1200);
    }
    
    @Test
    public void testSlowStart() {
        QuicPacer pacer = new QuicPacer();
        QuicCubicCongestionController cc = 
            new QuicCubicCongestionController(pacer);
        
        long initialCwnd = cc.getCongestionWindow();
        
        // 模拟收到 ACK
        cc.onAckReceived(1200, 100_000_000); // 1 个包, 100ms RTT
        
        // 慢启动阶段，窗口应增加
        assertTrue(cc.getCongestionWindow() > initialCwnd);
    }
    
    @Test
    public void testLossRecovery() {
        QuicPacer pacer = new QuicPacer();
        QuicCubicCongestionController cc = 
            new QuicCubicCongestionController(pacer);
        
        // 增加窗口到一定大小
        for (int i = 0; i < 100; i++) {
            cc.onAckReceived(1200, 100_000_000);
        }
        
        long beforeLoss = cc.getCongestionWindow();
        
        // 模拟丢包
        cc.onPacketLoss();
        
        // 窗口应减少到约 70% (β = 0.7)
        long afterLoss = cc.getCongestionWindow();
        assertTrue(afterLoss < beforeLoss);
        assertTrue(afterLoss > beforeLoss * 0.6);
        assertTrue(afterLoss < beforeLoss * 0.8);
    }
    
    @Test
    public void testCubicWindowFunction() {
        QuicPacer pacer = new QuicPacer();
        QuicCubicCongestionController cc = 
            new QuicCubicCongestionController(pacer);
        
        // 设置初始状态
        cc.onPacketLoss(); // 触发 CUBIC 状态初始化
        
        // 验证窗口增长是凹函数 (先慢后快再慢)
        // 在 t = k 时达到最大增长速率
        // 此处简化验证
        long prevCwnd = cc.getCongestionWindow();
        
        for (int i = 0; i < 10; i++) {
            cc.onAckReceived(1200, 100_000_000);
            long currentCwnd = cc.getCongestionWindow();
            assertTrue(currentCwnd >= prevCwnd);
            prevCwnd = currentCwnd;
        }
    }
}
```
