## 背景

### SuperWord 是什么？

SuperWord 是 C2 编译器的**自动向量化 (Auto-vectorization)** 优化，将循环中的标量操作转换为 SIMD 向量操作。

```
标量循环 (未优化):

for (int i = 0; i < n; i++) {
    c[i] = a[i] + b[i];  // 每次迭代处理 1 个元素
}

向量循环 (SuperWord 优化后):

// AVX-512: 每次处理 16 个 int
for (int i = 0; i < n; i += 16) {
    // 向量加法: 16 个元素并行
    vaddps(zmm0, zmm1, zmm2);
}
```

### 问题分析

```
SuperWord 问题场景:

1. 类型不匹配:
   for (int i = 0; i < n; i++) {
       double d = arr[i];  // int -> double 转换
       out[i] = d * 2.0;
   }
   问题: 类型转换导致向量化失败

2. 循环条件:
   for (int i = 0; i < n; i += 2) {  // 步长非 1
       c[i] = a[i] + b[i];
   }
   问题: 非标准步长导致向量化失败

3. 内存对齐:
   for (int i = offset; i < n; i++) {  // offset 未知
       c[i] = a[i] + b[i];
   }
   问题: 无法保证内存对齐，向量效率低
```

bool SuperWord::implementable(Node* n) {
    // 问题: 类型转换返回 false
    if (n->is_Convert()) {
        return false;  // 错误!
    }
    // ...
}

// 修复后: 支持类型转换向量化
bool SuperWord::implementable(Node* n) {
    if (n->is_Convert()) {
        // 检查是否是支持的类型转换
        int opcode = n->Opcode();
        switch (opcode) {
            case Op_ConvI2F:   // int -> float
            case Op_ConvI2D:   // int -> double
            case Op_ConvL2F:   // long -> float
            case Op_ConvL2D:   // long -> double
            case Op_ConvF2I:   // float -> int
            case Op_ConvF2D:   // float -> double
            case Op_ConvD2I:   // double -> int
            case Op_ConvD2F:   // double -> float
                return vector_width(n) > 0;
            default:
                return false;
        }
    }
    // ...
}

### 修复 2: 非标准步长支持

```cpp
// 文件: src/hotspot/share/opto/superword.cpp

// 修复前: 只支持步长 1
bool SuperWord::acceptable_loop() {
    CountedLoopNode* cl = _loop->as_CountedLoop();
    // 问题: 只接受 stride == 1
    return cl->stride_is_con() && cl->stride_con() == 1;
}

// 修复后: 支持更多步长
bool SuperWord::acceptable_loop() {
    CountedLoopNode* cl = _loop->as_CountedLoop();
    if (!cl->stride_is_con()) return false;

    int stride = cl->stride_con();

    // 支持 stride = 1, 2, 4, 8 等 2 的幂次
    if (stride >= 1 && is_power_of_2(stride)) {
        // 检查迭代次数是否足够
        int iv_count = cl->trip_count();
        int min_iters = vector_length() * stride;
        return iv_count >= min_iters;
    }

    return false;
}
```

### 修复 3: 未对齐内存访问

```cpp
// 文件: src/hotspot/share/opto/superword.cpp

// 新增: 对齐分析
class AlignmentAnalyzer {
public:
    // 分析内存访问的对齐情况
    static int analyze_alignment(Node* mem, const TypePtr* adr_type) {
        // 获取基地址
        Node* base = get_base_address(mem);
        if (base == nullptr) return 0;

        // 分析偏移量
        const TypePtr* base_type = _igvn.type(base)->isa_ptr();
        if (base_type == nullptr) return 0;

        // 计算对齐
        int offset = adr_type->offset();
        int alignment = 1;

        // 检查是否是数组访问
        if (base_type->isa_aryptr()) {
            // 数组元素大小
            int elem_size = base_type->is_aryptr()->elem()->size();
            alignment = elem_size;
        }

        return alignment;
    }

    // 生成对齐检查
    static Node* generate_alignment_check(
        Node* base, int required_alignment, PhaseIdealLoop* phase) {

        // 如果运行时对齐不确定，生成条件检查
        Node* mask = phase->igvn().intcon(required_alignment - 1);
        Node* aligned = phase->igvn().transform(
            new AndINode(base, mask));

        Node* zero = phase->igvn().intcon(0);
        Node* check = phase->igvn().transform(
            new CmpINode(aligned, zero));

        return phase->igvn().transform(
            new BoolNode(check, BoolTest::eq));
    }
};
```

### 新增: V-Transform (V-Transform)

```cpp
// 文件: src/hotspot/share/opto/superwordVTransform.cpp

/**
 * V-Transform: 向量变换优化
 * 
 * 将多个标量操作打包成一个向量操作，
 * 同时处理数据依赖和对齐问题。
 */
class VTransform : public ResourceObj {
private:
    VTransformGraph* _graph;
    VectorSet _packable;
    GrowableArray<VTransformNode*> _nodes;

public:
    // 构建向量变换图
    void build_graph() {
        // 1. 收集所有可向量化的节点
        collect_packable_nodes();

        // 2. 分析数据依赖
        analyze_dependencies();

        // 3. 构建打包集合
        build_packs();

        // 4. 生成向量代码
        generate_vector_code();
    }

    // 收集可打包的节点
    void collect_packable_nodes() {
        for (int i = 0; i < _graph->node_count(); i++) {
            Node* n = _graph->node(i);
            if (is_packable(n)) {
                _packable.set(i);
                _nodes.append(new VTransformNode(n));
            }
        }
    }

    // 分析数据依赖
    void analyze_dependencies() {
        for (int i = 0; i < _nodes.length(); i++) {
            VTransformNode* vtn = _nodes.at(i);
            Node* n = vtn->node();

            // 检查输入依赖
            for (uint j = 0; j < n->req(); j++) {
                Node* in = n->in(j);
                if (_packable.test(in->_idx)) {
                    vtn->add_dependency(in->_idx);
                }
            }
        }
    }

    // 构建打包集合
    void build_packs() {
        // 使用贪心算法构建最大打包集合
        VectorSet packed;
        
        for (int i = 0; i < _nodes.length(); i++) {
            if (packed.test(i)) continue;

            VTransformNode* vtn = _nodes.at(i);
            VTransformPack* pack = try_build_pack(vtn);

            if (pack != nullptr) {
                for (int j = 0; j < pack->size(); j++) {
                    packed.set(pack->at(j)->index());
                }
                _packs.append(pack);
            }
        }
    }

    // 尝试构建打包
    VTransformPack* try_build_pack(VTransformNode* seed) {
        VTransformPack* pack = new VTransformPack();
        pack->append(seed);

        // 向前/向后扩展打包
        extend_pack_forward(pack, seed);
        extend_pack_backward(pack, seed);

        // 检查打包是否有效
        if (pack->size() >= min_pack_size()) {
            return pack;
        }

        return nullptr;
    }
};
```

## 使用指南

### 可向量化的代码模式

```java
// ✅ 模式 1: 简单数组运算
for (int i = 0; i < n; i++) {
    c[i] = a[i] + b[i];
}

// ✅ 模式 2: 类型转换 (JDK 26 新支持)
for (int i = 0; i < n; i++) {
    float f = (float) intArray[i];
    floatArray[i] = f * 2.0f;
}

// ✅ 模式 3: 累加运算
float sum = 0;
for (int i = 0; i < n; i++) {
    sum += a[i];
}

// ❌ 模式 4: 条件分支 (无法向量化)
for (int i = 0; i < n; i++) {
    if (a[i] > 0) {
        c[i] = a[i];
    }
}

// ❌ 模式 5: 循环依赖 (无法向量化)
for (int i = 1; i < n; i++) {
    a[i] = a[i-1] + b[i];  // 依赖前一次迭代
}
```
