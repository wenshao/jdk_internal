# HttpClient HTTP/2 连接泄漏修复

> JDK-8326498 | Jaikiran Pai | JDK 26

---

## 概述

这是一个**严重的 bug 修复**，解决了 `java.net.http.HttpClient` 在使用 HTTP/2 时的连接池泄漏问题。该问题可能导致长时间运行的服务耗尽连接资源，最终阻塞或失败。

| 属性 | 值 |
|------|-----|
| **Issue** | [JDK-8326498](https://bugs.openjdk.org/browse/JDK-8326498) |
| **Commit** | `c19b12927d2` |
| **作者** | Jaikiran Pai |
| **目标版本** | JDK 26 (回移植到 21u, 17u) |
| **严重程度** | 🔴 严重 (P1) |
| **影响范围** | 所有 HTTP/2 客户端 |

---

## 问题描述

### 问题场景

```
时间线:
┌──────────────────────────────────────────────────────────────┐
│ T0: 创建 HttpClient                                          │
│     ↓                                                        │
│ T1: 发送 HTTP/2 请求 → 建立连接                              │
│     ↓                                                        │
│ T2: 请求完成，响应返回                                       │
│     ↓                                                        │
│ T3: 连接未正确释放 (BUG!)                                    │
│     ↓                                                        │
│ T4: 重复请求... 连接池逐渐填满                               │
│     ↓                                                        │
│ T5: 连接池耗尽                                               │
│     ↓                                                        │
│ T6: 新请求阻塞或失败 (OOM/Timeout)                           │
└──────────────────────────────────────────────────────────────┘
```

### 根因分析

```java
// 问题代码 (简化版)
// 文件: src/java.net.http/share/classes/jdk/internal/net/http/Http2Connection.java

class Http2Connection {
    
    void close() {
        // 问题: 某些异常路径未调用此方法
        if (state == ACTIVE) {
            sendGoAway();
            // 缺少: 从连接池移除!
        }
    }
    
    void handleStreamError(Stream stream, Throwable t) {
        // 问题: 错误路径未正确清理连接
        stream.close();
        // 缺少: 连接池清理
    }
    
    void onProtocolError() {
        // 问题: 协议错误处理不完整
        sendGoAway(ErrorCode.PROTOCOL_ERROR);
        // 缺少: 状态更新和连接池移除
    }
}
```

### 触发条件

| 条件 | 可能性 | 描述 |
|------|--------|------|
| 服务器关闭连接 | 高 | 服务器主动关闭，客户端未感知 |
| 网络中断 | 中 | 连接断开，异常处理不完整 |
| 超时 | 中 | 请求超时后连接状态不一致 |
| 并发请求 | 高 | 高并发下竞态条件 |
| 流错误 | 中 | 单个流错误影响整个连接 |

package jdk.internal.net.http;

/**
 * HTTP/2 连接终止原因枚举
 * 用于追踪和诊断连接关闭原因
 */
public enum Http2TerminationCause {
    
    /** 正常关闭 - 客户端主动关闭 */
    GRACEFUL_SHUTDOWN("Graceful shutdown"),
    
    /** 流错误 - 单个流出错，连接可能继续 */
    STREAM_ERROR("Stream error"),
    
    /** 连接错误 - 严重错误，必须关闭连接 */
    CONNECTION_ERROR("Connection error"),
    
    /** 空闲超时 - 连接空闲超时 */
    IDLE_TIMEOUT("Idle timeout"),
    
    /** 收到 GOAWAY - 服务器发送 GOAWAY 帧 */
    GOAWAY_RECEIVED("GOAWAY received"),
    
    /** 协议错误 - HTTP/2 协议违规 */
    PROTOCOL_ERROR("Protocol error"),
    
    /** 流被拒绝 - 服务器拒绝创建流 */
    REFUSED_STREAM("Refused stream"),
    
    /** 内部错误 - 客户端内部错误 */
    INTERNAL_ERROR("Internal error"),
    
    /** 对端关闭 - TCP 连接被对端关闭 */
    PEER_CLOSED("Peer closed connection"),
    
    /** SSL 错误 - TLS/SSL 层错误 */
    SSL_ERROR("SSL/TLS error");
    
    private final String description;
    
    Http2TerminationCause(String description) {
        this.description = description;
    }
    
    public String getDescription() {
        return description;
    }
}

### 修复: 连接终止逻辑

```java
// 文件: src/java.net.http/share/classes/jdk/internal/net/http/Http2Connection.java

class Http2Connection {
    
    private volatile State state = State.ACTIVE;
    private final Object closeLock = new Object();
    
    /**
     * 统一的连接终止方法
     * 确保所有路径都正确清理资源
     */
    private void terminate(Http2TerminationCause cause) {
        synchronized (closeLock) {
            if (state == State.CLOSED) {
                return; // 已关闭，避免重复
            }
            state = State.CLOSED;
        }
        
        // 1. 记录终止原因 (用于诊断)
        Log.logTrace("HTTP/2 connection terminating: {0}", cause);
        
        // 2. 发送 GOAWAY 帧 (如果需要)
        if (shouldSendGoAway(cause)) {
            try {
                sendGoAway(getGoAwayErrorCode(cause));
            } catch (IOException e) {
                Log.logTrace("Failed to send GOAWAY: {0}", e);
            }
        }
        
        // 3. 关闭所有活动流
        closeAllStreams(cause);
        
        // 4. 【关键修复】从连接池移除
        connectionPool().remove(this);
        
        // 5. 关闭底层 TCP 连接
        closeUnderlyingConnection();
        
        // 6. 通知等待的请求
        notifyPendingRequests(cause);
    }
    
    /**
     * 判断是否需要发送 GOAWAY
     */
    private boolean shouldSendGoAway(Http2TerminationCause cause) {
        return switch (cause) {
            case GRACEFUL_SHUTDOWN, CONNECTION_ERROR, PROTOCOL_ERROR -> true;
            case GOAWAY_RECEIVED, PEER_CLOSED, SSL_ERROR -> false;
            default -> state != State.CLOSING;
        };
    }
    
    /**
     * 将终止原因映射到 HTTP/2 错误码
     */
    private int getGoAwayErrorCode(Http2TerminationCause cause) {
        return switch (cause) {
            case PROTOCOL_ERROR -> ErrorCode.PROTOCOL_ERROR.code;
            case INTERNAL_ERROR -> ErrorCode.INTERNAL_ERROR.code;
            case REFUSED_STREAM -> ErrorCode.REFUSED_STREAM.code;
            default -> ErrorCode.NO_ERROR.code;
        };
    }
    
    /**
     * 关闭所有活动流
     */
    private void closeAllStreams(Http2TerminationCause cause) {
        List<Stream<?>> streamsToClose;
        synchronized (streams) {
            streamsToClose = new ArrayList<>(streams.values());
            streams.clear();
        }
        
        for (Stream<?> stream : streamsToClose) {
            try {
                stream.close(cause);
            } catch (Exception e) {
                Log.logTrace("Error closing stream: {0}", e);
            }
        }
    }
    
    /**
     * 通知等待中的请求
     */
    private void notifyPendingRequests(Http2TerminationCause cause) {
        IOException error = new IOException("Connection closed: " + cause);
        
        synchronized (pendingRequests) {
            for (CompletableFuture<?> future : pendingRequests.values()) {
                future.completeExceptionally(error);
            }
            pendingRequests.clear();
        }
    }
}
```

### 修复: 错误处理路径

```java
// 文件: src/java.net.http/share/classes/jdk/internal/net/http/Http2Connection.java

class Http2Connection {
    
    /**
     * 处理流错误
     */
    void handleStreamError(int streamId, ErrorCode errorCode) {
        if (errorCode == ErrorCode.REFUSED_STREAM) {
            // 流被拒绝，可能是服务器过载
            terminate(Http2TerminationCause.REFUSED_STREAM);
        } else if (isConnectionLevelError(errorCode)) {
            // 连接级别错误
            terminate(Http2TerminationCause.CONNECTION_ERROR);
        } else {
            // 仅关闭单个流
            Stream<?> stream = streams.get(streamId);
            if (stream != null) {
                stream.close(Http2TerminationCause.STREAM_ERROR);
            }
        }
    }
    
    /**
     * 处理收到 GOAWAY
     */
    void onGoAwayReceived(GoAwayFrame goAway) {
        int lastStreamId = goAway.getLastStreamId();
        int errorCode = goAway.getErrorCode();
        
        Log.logTrace("GOAWAY received: lastStreamId={0}, errorCode={1}",
                     lastStreamId, errorCode);
        
        // 标记正在关闭
        synchronized (closeLock) {
            state = State.CLOSING;
        }
        
        // 关闭超过 lastStreamId 的流
        closeStreamsAbove(lastStreamId);
        
        // 不立即终止，允许完成进行中的请求
        // 实际终止在所有流完成后或超时时
    }
    
    /**
     * 处理连接空闲超时
     */
    void onIdleTimeout() {
        Log.logTrace("HTTP/2 connection idle timeout");
        terminate(Http2TerminationCause.IDLE_TIMEOUT);
    }
    
    /**
     * 处理底层连接关闭
     */
    void onConnectionClosed(Throwable cause) {
        Log.logTrace("Underlying connection closed: {0}", cause);
        terminate(Http2TerminationCause.PEER_CLOSED);
    }
}
```

### 修复: 连接池管理

```java
// 文件: src/java.net.http/share/classes/jdk/internal/net/http/ConnectionPool.java

class ConnectionPool {
    
    private final Map<PoolKey, List<HttpConnection>> pool = new ConcurrentHashMap<>();
    private final Object poolLock = new Object();
    
    /**
     * 从连接池移除连接
     * 确保连接被正确清理
     */
    void remove(HttpConnection conn) {
        if (conn == null) return;
        
        PoolKey key = getPoolKey(conn);
        
        synchronized (poolLock) {
            List<HttpConnection> connections = pool.get(key);
            if (connections != null) {
                boolean removed = connections.remove(conn);
                if (removed) {
                    Log.logTrace("Removed connection from pool: {0}", conn);
                    
                    // 如果列表为空，移除整个 key
                    if (connections.isEmpty()) {
                        pool.remove(key);
                    }
                }
            }
        }
        
        // 确保连接被关闭
        try {
            conn.close();
        } catch (IOException e) {
            Log.logTrace("Error closing connection: {0}", e);
        }
    }
    
    /**
     * 清理过期连接
     */
    void cleanupExpired() {
        long now = System.currentTimeMillis();
        
        synchronized (poolLock) {
            for (List<HttpConnection> connections : pool.values()) {
                connections.removeIf(conn -> {
                    if (conn.isExpired(now)) {
                        Log.logTrace("Removing expired connection: {0}", conn);
                        return true;
                    }
                    return false;
                });
            }
        }
    }
}

// 文件: test/jdk/java/net/http/BurstyRequestsTest.java

/*
 * @test
 * @bug 8326498
 * @summary Test HTTP/2 connection leak with bursty requests
 * @run testng/othervm BurstyRequestsTest
 */

import java.net.URI;
import java.net.http.*;
import java.util.concurrent.*;
import java.util.concurrent.atomic.AtomicInteger;
import org.testng.annotations.Test;
import static org.testng.Assert.*;

public class BurstyRequestsTest {
    
    private static final int NUM_REQUESTS = 1000;
    private static final int CONCURRENCY = 10;
    
    @Test
    public void testConnectionNotLeaked() throws Exception {
        HttpClient client = HttpClient.newBuilder()
            .version(HttpClient.Version.HTTP_2)
            .connectTimeout(Duration.ofSeconds(10))
            .build();
        
        ExecutorService executor = Executors.newFixedThreadPool(CONCURRENCY);
        CountDownLatch latch = new CountDownLatch(NUM_REQUESTS);
        AtomicInteger successCount = new AtomicInteger(0);
        AtomicInteger failCount = new AtomicInteger(0);
        
        String url = "https://http2.example.org/api/test";
        
        for (int i = 0; i < NUM_REQUESTS; i++) {
            final int requestId = i;
            executor.submit(() -> {
                try {
                    HttpRequest request = HttpRequest.newBuilder()
                        .uri(URI.create(url + "?id=" + requestId))
                        .timeout(Duration.ofSeconds(30))
                        .build();
                    
                    HttpResponse<String> response = client.send(
                        request, 
                        HttpResponse.BodyHandlers.ofString()
                    );
                    
                    if (response.statusCode() == 200) {
                        successCount.incrementAndGet();
                    }
                } catch (Exception e) {
                    failCount.incrementAndGet();
                    System.err.println("Request " + requestId + " failed: " + e);
                } finally {
                    latch.countDown();
                }
            });
        }
        
        // 等待所有请求完成
        boolean completed = latch.await(5, TimeUnit.MINUTES);
        assertTrue(completed, "All requests should complete within timeout");
        
        // 验证成功率
        double successRate = (double) successCount.get() / NUM_REQUESTS;
        System.out.println("Success rate: " + (successRate * 100) + "%");
        assertTrue(successRate > 0.95, "Success rate should be > 95%");
        
        // 验证连接池状态
        // 由于 API 限制，我们通过内存使用间接验证
        long usedMemory = Runtime.getRuntime().totalMemory() 
                        - Runtime.getRuntime().freeMemory();
        System.out.println("Memory used: " + (usedMemory / 1024 / 1024) + " MB");
        
        executor.shutdown();
    }
    
    @Test
    public void testConnectionReuse() throws Exception {
        HttpClient client = HttpClient.newBuilder()
            .version(HttpClient.Version.HTTP_2)
            .build();
        
        String url = "https://http2.example.org/";
        
        // 发送多个请求到同一主机
        for (int i = 0; i < 10; i++) {
            HttpRequest request = HttpRequest.newBuilder()
                .uri(URI.create(url))
                .build();
            
            HttpResponse<String> response = client.send(
                request, 
                HttpResponse.BodyHandlers.ofString()
            );
            
            assertEquals(response.statusCode(), 200);
            assertEquals(response.version(), HttpClient.Version.HTTP_2);
        }
        
        // 验证连接被复用 (通过 HTTP/2 流 ID)
        // 第 10 个请求应该使用相同的连接
    }
    
    @Test
    public void testConnectionClosedOnError() throws Exception {
        HttpClient client = HttpClient.newBuilder()
            .version(HttpClient.Version.HTTP_2)
            .build();
        
        // 请求会触发服务器错误的 URL
        String errorUrl = "https://http2.example.org/error";
        
        HttpRequest request = HttpRequest.newBuilder()
            .uri(URI.create(errorUrl))
            .build();
        
        try {
            client.send(request, HttpResponse.BodyHandlers.ofString());
            fail("Should throw exception");
        } catch (Exception expected) {
            // 预期异常
        }
        
        // 验证后续请求仍能正常工作
        HttpRequest normalRequest = HttpRequest.newBuilder()
            .uri(URI.create("https://http2.example.org/"))
            .build();
        
        HttpResponse<String> response = client.send(
            normalRequest, 
            HttpResponse.BodyHandlers.ofString()
        );
        
        assertEquals(response.statusCode(), 200);
    }
}

### 检测连接泄漏

// 使用 JMX 监控连接池
import java.lang.management.*;
import jdk.management.jfr.*;

public class ConnectionLeakMonitor {
    
    public static void main(String[] args) throws Exception {
        // 方法 1: 使用 JFR 事件
        Configuration config = Configuration.getConfiguration("default");
        Recording recording = new Recording(config);
        recording.enable("jdk.HttpClientRequest");
        recording.enable("jdk.HttpClientResponse");
        recording.start();
        
        // 方法 2: 定期打印线程堆栈
        Thread monitor = new Thread(() -> {
            while (true) {
                try {
                    Thread.sleep(60000); // 每分钟
                    
                    // 统计 HttpClient 相关线程
                    int httpThreads = 0;
                    for (ThreadInfo info : ManagementFactory.getThreadMXBean()
                            .dumpAllThreads(false, false)) {
                        if (info.getThreadName().contains("HttpClient")) {
                            httpThreads++;
                        }
                    }
                    
                    System.out.println("HttpClient threads: " + httpThreads);
                    
                    // 如果线程数异常增长，可能有泄漏
                    if (httpThreads > 100) {
                        System.err.println("WARNING: Too many HttpClient threads!");
                    }
                } catch (InterruptedException e) {
                    break;
                }
            }
        });
        monitor.setDaemon(true);
        monitor.start();
    }
}
```

// 方案 1: 定期重建 HttpClient
public class HttpClientManager {
    private volatile HttpClient client;
    private final ScheduledExecutorService scheduler = 
        Executors.newSingleThreadScheduledExecutor();
    
    public HttpClientManager() {
        recreateClient();
        // 每小时重建客户端
        scheduler.scheduleAtFixedRate(
            this::recreateClient, 1, 1, TimeUnit.HOURS);
    }
    
    private synchronized void recreateClient() {
        HttpClient old = client;
        client = HttpClient.newBuilder()
            .version(HttpClient.Version.HTTP_2)
            .build();
        // 旧客户端会被 GC
    }
    
    public HttpClient getClient() {
        return client;
    }
}

// 方案 2: 使用连接池大小限制

    // 需要通过系统属性或自定义实现
    .build();
