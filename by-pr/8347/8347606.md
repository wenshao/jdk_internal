## 背景

### 什么是 ML-KEM？

ML-KEM (原名 Kyber) 是 NIST 标准化的**后量子密钥封装机制**，用于安全地交换密钥。

```
密钥封装机制 (KEM):

┌─────────────────────────────────────────────────────────────┐
│  Alice (发送方)                     Bob (接收方)            │
│                                                             │
│  1. 生成密钥对  ←───────────────────────                   │
│     公钥 PK, 私钥 SK                                      │
│                                                             │
│  2. 封装                              ────────────→ PK     │
│     输入: PK                                             │
│     输出: 密文 CT, 共享密钥 K                            │
│                                                             │
│  3. 解封        ←──────────────────── CT                  │
│     输入: CT, SK                                         │
│     输出: 共享密钥 K                                     │
│                                                             │
│  结果: Alice 和 Bob 拥有相同的共享密钥 K                   │
└─────────────────────────────────────────────────────────────┘

用途: TLS 握手中的密钥交换
```

### 参数集

| 参数集 | 安全级别 | 公钥大小 | 密文大小 | 共享密钥大小 |
|--------|----------|----------|----------|--------------|
| ML-KEM-512 | 128-bit | 800 bytes | 768 bytes | 32 bytes |
| ML-KEM-768 | 192-bit | 1,184 bytes | 1,088 bytes | 32 bytes |
| ML-KEM-1024 | 256-bit | 1,568 bytes | 1,536 bytes | 32 bytes |

### ML-KEM 核心实现

```java
// 文件: src/java.base/share/classes/com/sun/crypto/provider/ML_KEM.java

package com.sun.crypto.provider;

/**
 * ML-KEM 核心实现
 * 基于 FIPS 203 标准
 */
final class ML_KEM {

    // 参数集
    static final int K_512 = 2;
    static final int K_768 = 3;
    static final int K_1024 = 4;

    // ML-KEM-768 参数
    static final int N = 256;           // 多项式维度
    static final int Q = 3329;          // 模数
    static final int K = 3;             // 维度
    static final int ETA_1 = 2;         // CBD 参数
    static final int ETA_2 = 2;         // CBD 参数
    static final int DU = 10;           // 压缩参数
    static final int DV = 4;            // 压缩参数

    /**
     * 密钥生成
     */
    static KeyPair generateKeyPair(int k) {
        // 1. 生成随机种子
        byte[] d = new byte[32];
        SecureRandom random = new SecureRandom();
        random.nextBytes(d);

        // 2. 生成矩阵 A 和向量 s, e
        int[][] A = generateMatrixA(d, k);
        int[][] s = sampleCBD(ETA_1, k);
        int[][] e = sampleCBD(ETA_1, k);

        // 3. 计算公钥 t = A * s + e
        int[][] t = matrixMultiply(A, s);
        t = vectorAdd(t, e);

        // 4. 编码公钥和私钥
        byte[] publicKey = encodePublicKey(t, d, k);
        byte[] privateKey = encodePrivateKey(s, k);

        return new KeyPair(
            new ML_KEMPublicKey(publicKey, k),
            new ML_KEMPrivateKey(privateKey, k)
        );
    }

    /**
     * 封装
     */
    static Encapsulated encapsulate(byte[] publicKey, int k) {
        // 1. 解码公钥
        int[][] t = decodePublicKey(publicKey, k);
        byte[] rho = extractRho(publicKey, k);

        // 2. 生成随机 r, e1, e2
        byte[] m = new byte[32];
        byte[] r = new byte[32];
        SecureRandom random = new SecureRandom();
        random.nextBytes(m);
        System.arraycopy(m, 0, r, 0, 32);

        // 3. 生成矩阵和向量
        int[][] A = generateMatrixA(rho, k);
        int[][] s = sampleCBD(ETA_1, k);
        int[][] e1 = sampleCBD(ETA_2, k);
        int[] e2 = sampleCBD1D(ETA_2);

        // 4. 计算 u = A^T * r + e1
        int[][] u = matrixMultiplyTranspose(A, s);
        u = vectorAdd(u, e1);

        // 5. 计算 v = t^T * r + e2 + Encode(m)
        int[] v = vectorDotProduct(t, s);
        v = vectorAdd(v, e2);
        v = vectorAdd(v, encodeMessage(m));

        // 6. 压缩并编码密文
        byte[] ciphertext = encodeCiphertext(u, v, k);

        // 7. 派生共享密钥
        byte[] K = kdf(m, ciphertext);

        return new Encapsulated(
            new SecretKeySpec(K, "AES"),
            ciphertext
        );
    }

    /**
     * 解封
     */
    static byte[] decapsulate(byte[] privateKey, byte[] ciphertext, int k) 
            throws DecapsulateException {
        
        // 1. 解码私钥和密文
        int[][] s = decodePrivateKey(privateKey, k);
        int[][] u = decodeU(ciphertext, k);
        int[] v = decodeV(ciphertext, k);

        // 2. 计算消息 m = Decode(v - s^T * u)
        int[] inner = vectorDotProduct(s, u);
        int[] m_raw = vectorSubtract(v, inner);
        byte[] m = decodeMessage(m_raw);

        // 3. 重封装验证
        byte[] expectedCT = reEncapsulate(m, k);
        if (!constantTimeCompare(ciphertext, expectedCT)) {
            // 返回假密钥 (避免侧信道攻击)
            return generateFakeKey(privateKey);
        }

        // 4. 派生共享密钥
        return kdf(m, ciphertext);
    }

        // ...
    }

    private static byte[] kdf(byte[] m, byte[] ciphertext) {
        // SHA3-256 密钥派生
        MessageDigest sha3 = MessageDigest.getInstance("SHA3-256");
        sha3.update(m);
        sha3.update(ciphertext);
        return sha3.digest();

    private static boolean constantTimeCompare(byte[] a, byte[] b) {
        if (a.length != b.length) return false;
        int result = 0;
```

### AVX-512 优化

```asm
// 文件: src/hotspot/cpu/x86/ml_kem_x86_64.asm

// ML-KEM NTT 变换 (AVX-512 优化)

.global ml_kem_ntt_avx512
ml_kem_ntt_avx512:
    // rdi: 多项式数组
    // rsi: 长度

    // 使用 AVX-512 处理 16 个系数
    vmovdqa64 zmm0, [rdi]          // 加载 16 个系数

    // NTT 蝶形运算
    // Cooley-Tukey 算法
    vpmullw zmm1, zmm0, zmm16      // 乘以旋转因子
    vpaddw zmm2, zmm0, zmm1        // 加法
    vpsubw zmm3, zmm0, zmm1        // 减法

    // 模约减 (使用 Barrett 约减)
    vpmulhuw zmm4, zmm2, [barrett_const]
    vpsrlw zmm4, zmm4, 16
    vpmullw zmm4, zmm4, [q_const]
    vpsubw zmm2, zmm2, zmm4

    vmovdqa64 [rdi], zmm2          // 存储结果

    ret
```

## 使用示例

### 基本 KEM 操作

```java
import javax.crypto.*;
import java.security.*;

public class ML_KEM_Example {

    public static void main(String[] args) throws Exception {
        // 1. 生成密钥对
        KeyPairGenerator kpg = KeyPairGenerator.getInstance("ML-KEM-768");
        KeyPair keyPair = kpg.generateKeyPair();

        // 2. Alice 封装
        KEM kem = KEM.getInstance("ML-KEM");
        KEM.Encapsulator enc = kem.newEncapsulator(keyPair.getPublic());
        KEM.Encapsulated encapsulated = enc.encapsulate(0, 32, "AES");

        byte[] ciphertext = encapsulated.ciphertext();
        SecretKey aliceKey = encapsulated.key();

        System.out.println("Ciphertext size: " + ciphertext.length + " bytes");
        // 输出: Ciphertext size: 1088 bytes (ML-KEM-768)

        // 3. Bob 解封
        KEM.Decapsulator dec = kem.newDecapsulator(keyPair.getPrivate());
        SecretKey bobKey = dec.decapsulate(ciphertext, 0, 32, "AES");

        // 4. 验证密钥相同
        System.out.println("Keys match: " + 
            MessageDigest.isEqual(aliceKey.getEncoded(), bobKey.getEncoded()));
        // 输出: Keys match: true
    }
}
```

### TLS 集成

```java
import javax.net.ssl.*;

public class TLS_ML_KEM_Example {

    public static void main(String[] args) throws Exception {
        // 创建 SSL 上下文
        SSLContext sslContext = SSLContext.getInstance("TLS");

        // 配置使用 ML-KEM 密钥交换
        SSLParameters params = new SSLParameters();
        params.setKeyExchangeGroups("x25519_mlkem768", "x25519");

        // 创建服务器
        SSLServerSocketFactory ssf = sslContext.getServerSocketFactory();
        SSLServerSocket serverSocket = 
            (SSLServerSocket) ssf.createServerSocket(8443);
        serverSocket.setSSLParameters(params);

        // 接受连接
        try (SSLSocket socket = (SSLSocket) serverSocket.accept()) {
            // 握手完成后，使用了 ML-KEM 密钥交换
            System.out.println("Negotiated suite: " + 
                socket.getSession().getCipherSuite());
        }
    }
}
```
